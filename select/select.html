参数说明

				selectUI.select($(".selectUI"),{
					selectClass: "mySel",//给selectUI的最外层添加class//默认不添加
					timer:100,//下拉时间,默认200ms
					isScroll:false,//鼠标在下拉框中时是否页面滚动
					maxHeight:225,//设置下拉框的最大高度，默认0不设置
					callBack : function(){//下拉框点击后的回调事件
						alert("当前选中值为"+$(".selectUI1").val());
					}
					
				});
				

				使用方法,在原始select上加class "selectUI"
				.调用selectUI.init();或
				.在select上添加另外class,selectUI.select($(".selectUI1"),{
					//这里是参数
				})
