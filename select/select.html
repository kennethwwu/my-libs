Javascript插件

		selectUI.select($(".selectUI1"),{
			selectClass: "mySel",
			timer:400,//下拉时间,默认400ms
			isScroll:false,//鼠标在下拉框中时页面不滚动
			maxHeight:225,
			callBack : function(){
				alert("当前选中值为"+$(".selectUI1").val());
			}
			
		});
		
			//select禁止
			selectUI.select($(".selectUI2"));
		
			selectUI.select($(".selectUI3"),{
				timer:100,下拉时间100ms
				maxHeight:235,//下拉框的最大高度为235px
				isScroll:true,//鼠标在下拉框中时页面滚动
				callBack: function(){
					alert("当前选中值为"+$(".selectUI3").val());
				}
			});

参数说明

				selectUI.select($(".selectUI"),{
					selectClass: "mySel",//给selectUI的最外层添加class//默认不添加
					timer:200,//下拉时间,默认200ms
					isScroll:false,//鼠标在下拉框中时页面是否滚动,默认不滚动
					maxHeight:225,//设置下拉框的最大高度，默认0不设置
					callBack : function(){//下拉框点击后的回调事件
						alert("当前选中值为"+$(".selectUI1").val());
					}
					
				});
				

				
使用方法

				导入selectUI.css和导入selectUI.js(依赖jquery),在原始select上加class "selectUI"
				.调用selectUI.init();或
				.在select上添加另外class,selectUI.select($(".selectUI1"),{
					//这里是参数
				})
